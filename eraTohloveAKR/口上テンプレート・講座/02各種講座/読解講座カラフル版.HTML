【4. IF-ENDIF의 대응 독해】
;-------------------------------------------------
;0,애무
;-------------------------------------------------

　　　　CALL TRAIN_MESSAGE

　　　　　　　　;함락이 끝난 상태
　　　　　　　　IF (TALENT:3 || TALENT:4 || TALENT:5 || TALENT:6 || TALENT:7 || TALENT:8) && (CFLAG:300 < 4 || FLAG:7 == 2)

　　　　　　　　　　　　PRINTFORMW 【등 째네】

　　　　　　　　　　　　CFLAG:300 = 4
　　　　　　　　;굴복각인Lv3

　　　　ENDIF
ENDIF
